\section{Introducción a Taylor}
\begin{frame}

 \[f(x) = f(a)
  + \frac{f'(a)}{1!}(x - a)
  + \frac{f^{(2)}(a)}{2!}(x - a)^2
  + \cdots
  + \frac{f^{(n)}(a)}{n!}(x - a)^n
  + R_n(f)\]
  
\[f(x) = \sum_{k=0}^n \frac{f^{(k)}(a)}{k!}(x - a)^k + R_n(f)\]

\[R_n(f) = \frac{f^{(n+1)}(\xi)}{(n+1)!} (x-a)^{n+1}\]

\[R_n(f) = \int_a^x \frac{f^{(n+1)} (t)}{n!} (x - t)^n \, dt\]

\end{frame}
